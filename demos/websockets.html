Web Sockets

//connect to our server
var socket = io.connect('http://thefutureisnow.nullempire.com:80');
//when we recieve a message, show it in the DOM
socket.on('receiveDemoMessage', function(message){
    //append to messages list
});
//when the document is ready, attach DOM events
$(function(){

    $('#message').on('keyup', function(e){
        //when the enter key is pressed, send a message
        if(e.keyCode === 13){
            var el = $(this);
            var message = el.val();
            socket.emit('sendDemoMessage', message);
            el.val('');
        }
    });
});
