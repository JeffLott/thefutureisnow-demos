Orientation

if(window.DeviceOrientationEvent){
    window.addEventListener('deviceorientation', function(e){
        console.log('left-right : ', e.gamma);
        console.log('front-back : ', e.beta);
        console.log('direction : ', e.alpha);
        $('#box').css('-webkit-transform','rotate('+e.gamma+'deg) rotate3d(1,0,0, -'+e.beta+'deg)');
    }, false);
}
